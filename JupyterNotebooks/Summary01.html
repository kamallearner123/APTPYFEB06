4Ô∏è‚É£ Function to check if a number is prime

def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True
print(is_prime(11))

5Ô∏è‚É£ Function to calculate factorial

def factorial(n):
    if n == 0 or n == 1:
        return 1
    return n * factorial(n - 1)
print(factorial(5))

1Ô∏è‚É£9Ô∏è‚É£ Find all prime numbers in a range

def primes_in_range(n):
    primes = []
    for num in range(2, n+1):
        if all(num % i != 0 for i in range(2, int(num ** 0.5) + 1)):
            primes.append(num)
    return primes
print(primes_in_range(50))

def calculator(a, b, op):
    if op == '+': return a + b
    if op == '-': return a - b
    if op == '*': return a * b
    if op == '/': return a / b if b != 0 else "Error"
print(calculator(10, 5, '+'))
